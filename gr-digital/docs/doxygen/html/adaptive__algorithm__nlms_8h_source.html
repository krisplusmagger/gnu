58 float magsq = real(dp);
59 float norm_step_size = d_step_size / magsq;
60
61 for (unsigned i = 0; i < num_taps; i++) {
62 taps[i] = conj(conj(taps[i]) + norm_step_size * in[i] * conj(error));
63 }
64 }
